Table Users {
  user_id INT [primary key]
  username nvarchar(30) [NOT NULL]
  password nvarchar(30) [NOT NULL]
}

Table Jokes {
  joke_id INT [pk]
  user_id INT
  rating INT
}

Table Comments {
  comment_id INT [pk]
  user_id INT
  joke_id INT
}
